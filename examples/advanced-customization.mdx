## Custom Selection Callbacks

Add callbacks to handle item selection events:

<CodeGroup>
```dart Selection Callbacks
class CustomSliceAnimatedList extends StatefulWidget {
  final Function(int index)? onItemSelected;
  final Function(ProfileData profile)? onProfileTapped;
  
  const CustomSliceAnimatedList({
    super.key,
    this.onItemSelected,
    this.onProfileTapped,
  });

  @override
  _CustomSliceAnimatedListState createState() => _CustomSliceAnimatedListState();
}

class _CustomSliceAnimatedListState extends State<CustomSliceAnimatedList>
    with TickerProviderStateMixin {
  
  void _selectItem(int index) {
    setState(() {
      _selectedIndex = index;
    });
    
    // Trigger callback when item is selected
    widget.onItemSelected?.call(index);
    widget.onProfileTapped?.call(_profiles[index]);
  }
  
  // Use in GestureDetector
  Widget _buildCarouselItem(BuildContext context, int index) {
    final double scale = _calculateItemScale(index);
    final double offsetX = (index - _selectedIndex) * _itemSpacing;

    return AnimatedPositioned(
      duration: _animationDuration,
      curve: Curves.easeInOut,
      left: MediaQuery.of(context).size.width / 2 - _itemWidth / 2 + offsetX,
      child: GestureDetector(
        onTap: () => _selectItem(index),
        child: AnimatedScale(
          scale: scale,
          duration: _animationDuration,
          curve: Curves.easeInOut,
          child: _buildProfileCard(context, _profiles[index], scale),
        ),
      ),
    );
  }
}
```
</CodeGroup>

## Custom Animation Curves

Apply different easing curves for unique motion effects:

<CodeGroup>
```dart Custom Curves
Widget _buildCarouselItem(BuildContext context, int index) {
  final double scale = _calculateItemScale(index);
  final double offsetX = (index - _selectedIndex) * _itemSpacing;

  return AnimatedPositioned(
    duration: _animationDuration,
    curve: Curves.elasticOut,  // Bouncy effect
    left: MediaQuery.of(context).size.width / 2 - _itemWidth / 2 + offsetX,
    child: GestureDetector(
      onTap: () => _selectItem(index),
      child: AnimatedScale(
        scale: scale,
        duration: _animationDuration,
        curve: Curves.easeOutBack,  // Overshoot effect
        child: _buildProfileCard(context, _profiles[index], scale),
      ),
    ),
  );
}
```
</CodeGroup>

<Info>
Popular curves to try:
- `Curves.easeInOut` - Smooth, balanced (default)
- `Curves.elasticOut` - Bouncy, playful
- `Curves.easeOutBack` - Slight overshoot
- `Curves.fastOutSlowIn` - Material Design standard
- `Curves.decelerate` - Quick start, slow finish
</Info>

class ProfileData {
  final String name;
  final String imageUrl;
  final String role;
  
  ProfileData({
    required this.name,
    required this.imageUrl,
    required this.role,
  });
}

class CustomSliceAnimatedList extends StatefulWidget {
  final Function(ProfileData profile)? onProfileTapped;
  
  const CustomSliceAnimatedList({super.key, this.onProfileTapped});

  @override
  _CustomSliceAnimatedListState createState() => _CustomSliceAnimatedListState();
}

class _CustomSliceAnimatedListState extends State<CustomSliceAnimatedList>
    with TickerProviderStateMixin {
  
  // Custom animation constants
  static const double _itemWidth = 180.0;
  static const double _itemSpacing = 220.0;
  static const double _maxScale = 1.2;
  static const double _minScale = 0.4;
  static const double _scaleFactor = 0.15;
  static const Duration _animationDuration = Duration(milliseconds: 450);
  static const Duration _buttonPulseDuration = Duration(milliseconds: 2000);
  
  final List<ProfileData> _profiles = [
    ProfileData(
      name: 'Alice Johnson',
      imageUrl: 'https://i.pravatar.cc/150?img=1',
      role: 'Senior Developer',
    ),
    ProfileData(
      name: 'Bob Smith',
      imageUrl: 'https://i.pravatar.cc/150?img=2',
      role: 'Product Manager',
    ),
    ProfileData(
      name: 'Carol Williams',
      imageUrl: 'https://i.pravatar.cc/150?img=3',
      role: 'UX Designer',
    ),
    ProfileData(
      name: 'David Brown',
      imageUrl: 'https://i.pravatar.cc/150?img=4',
      role: 'DevOps Engineer',
    ),
    ProfileData(
      name: 'Emma Davis',
      imageUrl: 'https://i.pravatar.cc/150?img=5',
      role: 'Data Scientist',
    ),
  ];
  
  int _selectedIndex = 2;
  late AnimationController _buttonAnimationController;
  late Animation<double> _buttonAnimation;
  
  @override
  void initState() {
    super.initState();
    _initializeButtonAnimation();
  }
  
  void _initializeButtonAnimation() {
    _buttonAnimationController = AnimationController(
      vsync: this,
      duration: _buttonPulseDuration,
    )..repeat(reverse: true);

    _buttonAnimation = Tween<double>(begin: 1.0, end: 1.3).animate(
      CurvedAnimation(
        parent: _buttonAnimationController,
        curve: Curves.easeInOut,
      ),
    );
  }
  
  @override
  void dispose() {
    _buttonAnimationController.dispose();
    super.dispose();
  }
  
  void _selectItem(int index) {
    setState(() {
      _selectedIndex = index;
    });
    widget.onProfileTapped?.call(_profiles[index]);
  }
  
  void _selectPreviousItem() {
    if (_selectedIndex > 0) {
      _selectItem(_selectedIndex - 1);
    }
  }
  
  void _selectNextItem() {
    if (_selectedIndex < _profiles.length - 1) {
      _selectItem(_selectedIndex + 1);
    }
  }
  
  double _calculateItemScale(int index) {
    final int distanceFromCenter = (index - _selectedIndex).abs();
    return (_maxScale - (distanceFromCenter * _scaleFactor)).clamp(
      _minScale,
      _maxScale,
    );
  }
  
  Widget _buildProfileCard(
    BuildContext context,
    ProfileData profile,
    bool isSelected,
  ) {
    return SizedBox(
      width: _itemWidth,
      child: Card(
        elevation: isSelected ? 12 : 4,
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(16),
          side: isSelected
              ? const BorderSide(color: Colors.blue, width: 3)
              : BorderSide.none,
        ),
        child: Padding(
          padding: const EdgeInsets.all(12.0),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              CircleAvatar(
                radius: 35,
                backgroundImage: NetworkImage(profile.imageUrl),
              ),
              const SizedBox(height: 8),
              Text(
                profile.name,
                style: TextStyle(
                  fontSize: 14,
                  fontWeight: isSelected ? FontWeight.bold : FontWeight.normal,
                  color: isSelected ? Colors.blue : Colors.white,
                ),
                textAlign: TextAlign.center,
                maxLines: 1,
                overflow: TextOverflow.ellipsis,
              ),
              const SizedBox(height: 4),
              Text(
                profile.role,
                style: TextStyle(
                  fontSize: 11,
                  color: Colors.grey[600],
                ),
                textAlign: TextAlign.center,
                maxLines: 1,
                overflow: TextOverflow.ellipsis,
              ),
            ],
          ),
        ),
      ),
    );
  }
  
  Widget _buildCarouselItem(BuildContext context, int index) {
    final double scale = _calculateItemScale(index);
    final double offsetX = (index - _selectedIndex) * _itemSpacing;
    final bool isSelected = index == _selectedIndex;

    return AnimatedPositioned(
      duration: _animationDuration,
      curve: Curves.easeInOut,
      left: MediaQuery.of(context).size.width / 2 - _itemWidth / 2 + offsetX,
      child: GestureDetector(
        onTap: () => _selectItem(index),
        child: AnimatedScale(
          scale: scale,
          duration: _animationDuration,
          curve: Curves.easeInOut,
          child: _buildProfileCard(context, _profiles[index], isSelected),
        ),
      ),
    );
  }
  
  Widget _buildNavigationButton({
    required IconData icon,
    required VoidCallback onPressed,
    required bool isLeft,
  }) {
    return AnimatedBuilder(
      animation: _buttonAnimationController,
      builder: (context, child) {
        return Transform.scale(
          scale: _buttonAnimation.value,
          child: Container(
            decoration: BoxDecoration(
              gradient: LinearGradient(
                colors: [
                  Colors.blue.shade700,
                  Colors.blue.shade900,
                ],
                begin: Alignment.topLeft,
                end: Alignment.bottomRight,
              ),
              shape: BoxShape.circle,
              boxShadow: [
                BoxShadow(
                  color: Colors.blue.withOpacity(0.5),
                  blurRadius: 10,
                  spreadRadius: 2,
                ),
              ],
            ),
            child: IconButton(
              icon: Icon(icon, color: Colors.white),
              onPressed: onPressed,
              iconSize: 32,
            ),
          ),
        );
      },
    );
  }
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: <Widget>[
        SizedBox(
          height: 200.0,
          width: MediaQuery.of(context).size.width,
          child: Stack(
            alignment: Alignment.center,
            children: [
              ...List.generate(
                _profiles.length,
                (index) => _buildCarouselItem(context, index),
              ),
              if (_selectedIndex > 0)
                Positioned(
                  left: 20,
                  child: _buildNavigationButton(
                    icon: Icons.arrow_back_ios,
                    onPressed: _selectPreviousItem,
                    isLeft: true,
                  ),
                ),
              if (_selectedIndex < _profiles.length - 1)
                Positioned(
                  right: 20,
                  child: _buildNavigationButton(
                    icon: Icons.arrow_forward_ios,
                    onPressed: _selectNextItem,
                    isLeft: false,
                  ),
                ),
            ],
          ),
        ),
      ],
    );
  }
}
